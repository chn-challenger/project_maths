\begin{minipage}[t]{1.0000\textwidth}
\begin{align*}
\intertext{1.\hspace{30pt}Justin is sixty three years older than his granddaughter. Fifteen years from now, Justin will be four times as old as his granddaughter. How old is his granddaughter now?}&\text{Now}&&&&\text{15 years from now}&\\
&\text{granddaughter}\hspace{10pt}x&&&&\text{granddaughter}\hspace{10pt}x+15&\\
&\text{Justin}\hspace{10pt}x + 63&&&&\text{Justin}\hspace{10pt}x + 63 + 15&\\
&&\text{Justin} &= \text{4} \times \text{granddaughter}&\\
&&x+78&=4\left(x+15\right)&\\
&&x+78&=4x+60&\\
&&78-60&=4x-x&\\
&&18&=3x&\\
&&\cfrac{18}{3}&=x&\\
&&6&=x&\\
&Answer:&\text{granddaughter now} &= 6
\end{align*}
\end{minipage}
\vspace{10 mm}

\noindent
\begin{minipage}[t]{1.0000\textwidth}
\begin{align*}
\intertext{2.\hspace{30pt}Ken is twenty four years younger than his aunt. Seven years from now, his aunt will be twice as old as him. How old is Ken now?}&\text{Now}&&&&\text{7 years from now}&\\
&\text{Ken}\hspace{10pt}x&&&&\text{Ken}\hspace{10pt}x+7&\\
&\text{aunt}\hspace{10pt}x + 24&&&&\text{aunt}\hspace{10pt}x + 24 + 7&\\
&&\text{aunt} &= \text{2} \times \text{Ken}&\\
&&x+31&=2\left(x+7\right)&\\
&&x+31&=2x+14&\\
&&31-14&=2x-x&\\
&&17&=x&\\
&Answer:&\text{Ken now} &= 17
\end{align*}
\end{minipage}
\vspace{10 mm}

\noindent
\begin{minipage}[t]{1.0000\textwidth}
\begin{align*}
\intertext{3.\hspace{30pt}In three years time, Ken will be seven times as old as his granddaughter. Twelve years from now, Ken will be four times as old as his granddaughter. How old is his granddaughter now?}&\text{3 years from now}&&&&\text{12 years from now}&\\
&\text{granddaughter}\hspace{10pt}x&&&&\text{granddaughter}\hspace{10pt}x+9&\\
&\text{Ken}\hspace{10pt}7x&&&&\text{Ken}\hspace{10pt}7x+9&\\
&&\text{Ken} &= \text{4} \times \text{granddaughter}&\\
&&7x+9&=4\left(x+9\right)&\\
&&7x+9&=4x+36&\\
&&7x-4x&=36-9&\\
&&3x&=27&\\
&&x&=\cfrac{27}{3}&\\
&&x&=9&\\
&&\text{granddaughter now} &= \text{granddaughter 3 years from now} - 3\\
&&\text{granddaughter now} &= 9 - 3\\
&Answer:&\text{granddaughter now} &= 6
\end{align*}
\end{minipage}
\vspace{10 mm}

\begin{align*}
\intertext{4.\hspace{30pt}In five years time, Beth's grandfather will be three times as old as Beth. Twenty nine years from now, Beth's grandfather will be twice as old as Beth. How old is Beth now?}&\text{5 years from now}&&&&\text{29 years from now}&\\
&\text{Beth}\hspace{10pt}x&&&&\text{Beth}\hspace{10pt}x+24&\\
&\text{grandfather}\hspace{10pt}3x&&&&\text{grandfather}\hspace{10pt}3x+24&\\
&&\text{grandfather} &= \text{2} \times \text{Beth}&\\
&&3x+24&=2\left(x+24\right)&\\
&&3x+24&=2x+48&\\
&&3x-2x&=48-24&\\
&&x&=24&\\
&&\text{Beth now} &= \text{Beth 5 years from now} - 5\\
&&\text{Beth now} &= 24 - 5\\
&Answer:&\text{Beth now} &= 19
\end{align*}
